    </div>

    <script>
        // Sample JSON data
        const data = /* JSON data here */;
        // Store the data in localStorage to allow for editing and persistence
        if (!localStorage.getItem("studentData")) {
            localStorage.setItem("studentData", JSON.stringify(data));
        }

        function fetchResults() {
            const rollNumber = document.getElementById("rollNumber").value;
            const year = document.getElementById("year").value;
            
            if (!rollNumber || !year) {
                alert("Please fill in both fields.");
                return;
            }
            
            const studentData = JSON.parse(localStorage.getItem("studentData")).find(
                student => student['Roll No.'] == rollNumber && student['Year'] === year
            );
            
            if (studentData) {
                displayResults(studentData);
            } else {
                alert("No data found for this roll number and year.");
            }
        }

        function displayResults(student) {
            const resultTable = document.getElementById("resultTable");
            let tableHTML = `<table>`;

            Object.keys(student).forEach((key) => {
                if (key !== 'Roll No.' && key !== 'Year') {
                    tableHTML += `
                        <tr>
                            <th>${key}</th>
                            <td>
                                <span class="editable" onclick="editField('${key}')">${student[key]}</span>
                                <input type="text" class="edit-input" id="${key}-input" value="${student[key]}" style="display:none;">
                            </td>
                        </tr>
                    `;
                }
            });

            tableHTML += `</table><button class="save-btn" onclick="saveChanges()">Save Changes</button>`;
            resultTable.innerHTML = tableHTML;

            document.getElementById("page1").style.display = "none";
            document.getElementById("page2").style.display = "block";
        }

        function editField(field) {
            document.getElementById(`${field}-input`).style.display = "inline";
            document.querySelector(`span[onclick="editField('${field}')"]`).style.display = "none";
        }

        function saveChanges() {
            const updatedData = {};
            const inputs = document.querySelectorAll(".edit-input");

            inputs.forEach(input => {
                const field = input.id.replace("-input", "");
                updatedData[field] = input.value;
            });

            // Update the student data in localStorage
            let studentData = JSON.parse(localStorage.getItem("studentData"));
            studentData = studentData.map(student => {
                if (student["Roll No."] == document.getElementById("rollNumber").value && 
                    student["Year"] === document.getElementById("year").value) {
                    return { ...student, ...updatedData };
                }
                return student;
            });
            localStorage.setItem("studentData", JSON.stringify(studentData));

            alert("Changes saved successfully!");
            fetchResults(); // Refresh the displayed data
        }

        function goBack() {
            document.getElementById("page1").style.display = "block";
            document.getElementById("page2").style.display = "none";
        }
    </script>
